## 2. arg parse pattern [REUSE]

from `patent.priors.search.sh`:

```bash
parse_args() {
  QUERY=""
  LIMIT=25
  SINCE=""
  UNTIL=""

  while [[ $# -gt 0 ]]; do
    case $1 in
      --skill|--repo|--role) shift 2 ;;  # rhachet passes these
      --query) QUERY="$2"; shift 2 ;;
      --limit) LIMIT="$2"; shift 2 ;;
      --since) SINCE="$2"; shift 2 ;;
      --until) UNTIL="$2"; shift 2 ;;
      --help|-h) usage; exit 0 ;;
      *) echo "unknown argument: $1" >&2; exit 2 ;;
    esac
  done
}
```

**fit**: direct reuse. permit.search needs --address, --postal, --limit, --since, --until.

## 3. keyrack credential retrieval [EXTEND]

from `patent.priors.search.sh`:

```bash
get_api_key() {
  local keyrack_json
  keyrack_json=$(rhx keyrack get --key ehmpathy.test.USPTO_ODP_API_KEY --json 2>/dev/null) || {
    print_blocked "keyrack failed" "run: rhx keyrack unlock --owner ehmpath --env test"
    exit 2
  }

  local status key
  status=$(echo "$keyrack_json" | jq -r '.status')
  key=$(echo "$keyrack_json" | jq -r '.value // empty')

  if [[ "$status" != "unlocked" ]] || [[ -z "$key" ]]; then
    print_blocked "API key not available" "run: rhx keyrack unlock --owner ehmpath --env test"
    exit 2
  fi

  echo "$key"
}
```

## 4. retry with backoff [REUSE]

from `patent.priors.search.sh`:

```bash
fetch_with_retry() {
  local url="$1"
  local max_attempts=3
  local attempt=1

  while [[ $attempt -le $max_attempts ]]; do
    local response http_code body
    response=$(curl -s -w "\n%{http_code}" -H "X-Api-Key: $API_KEY" "$url")
    http_code=$(echo "$response" | tail -1)
    body=$(echo "$response" | sed '$d')

    if [[ "$http_code" == "200" ]]; then
      echo "$body"
      return 0
    elif [[ "$http_code" == "429" ]]; then
      local wait=$((attempt * 2))
      sleep "$wait"
      ((attempt++))
    else
      echo "HTTP $http_code: $body" >&2
      return 1
    fi
  done

  echo "max retries exceeded" >&2
  return 1
}
```

**fit**: direct reuse for permit API calls or scrape retries.

  local is_last="$2"
  local key="$3"

parse_search_response() {
  local json="$1"

  # extract total count
  local total
  total=$(echo "$json" | jq -r '.count // 0')
